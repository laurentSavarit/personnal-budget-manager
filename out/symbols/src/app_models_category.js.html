1 const pool = require("../db");
  2 /**

  7 /**
  8  * @typedef categoryPost

 11 class Category {
 12 
 13 
 14     constructor(obj={}){
 15         for(const prop in obj){
 16             this[prop] = obj[prop];
 17         };
 18     };
 19 
 20     /**
 21      * Find all categories
 22      * @static
 23      * @async
 24      * @returns {[Category]} an array of category instance
 25      */
 26     static async findAll(){
 27 
 28         try{
 29         const sqlQuery = "SELECT * FROM category;";
 30         const {rows} = await pool.query(sqlQuery);
 31 
 32         return rows.map(row=>new this(row));
 33 
 34         }catch(err){
 35             if(err.detail){
 36                 throw new Error(err.detail);
 37             }else{
 38                 throw err;
 39             }
 40         }
 41     };
 42 
 43     /**
 44      * Find one category by id in BDD
 45      * @param {integer} id 
 46      * @static
 47      * @async
 48      * @returns {Category} return one category instance
 49      */
 50     static async findOne(id){
 51 
 52         try{
 53         const sqlQuery = {
 54             text: "SELECT * FROM category WHERE id=$1;",
 55             values: [id]
 56         };
 57         const {rows} = await pool.query(sqlQuery);
 58 
 59         return rows[0] ? new this(rows[0]) : new Error("id not found");
 60 
 61         }catch(err){
 62             if(err.detail){
 63                 throw new Error(err.detail);
 64             }else{
 65                 throw err;
 66             }
 67         }
 68     };
 69 
 70     /**
 71      * Save : insert or update in BDD
 72      * @async
 73      * @returns {Category} if create return an instance of Category
 74      */
 75     async save(){
 76         try{
 77 
 78             const sqlQuery = {
 79                 text:"SELECT id FROM insert_category($1);",
 80                 values:[this]
 81             };
 82 
 83             if(this.id){
 84                 sqlQuery.text = "SELECT update_category($1);";
 85             }
 86 
 87             const {rows} = await pool.query(sqlQuery);
 88 
 89             if(!this.id){
 90                 this.id = rows[0].id;
 91                 return this;
 92             }else{
 93                 return;
 94             }
 95 
 96         }catch(err){
 97             if(err.detail){
 98                 throw new Error(err.detail);
 99             }else{
100                 throw err;
101             }
102         }
103     };
104 
105     /**

107      * @param {integer} id the if of category
108      * @static
109      * @async
110      * @returns {boolean} return true or false
111      */
112     static async delete(id){
113         try{
114 
115           const sqlQuery = {
116               text: "DELETE FROM category WHERE id=$1",
117               values:[parseInt(id,10)]
118           };
119           
120           const {rowCount} = await pool.query(sqlQuery);
121 
122           return rowCount ? true : false;
123 
124         }catch(err){
125             if(err.detail){
126                 throw new Error(err.detail);
127             }else{
128                 throw err;
129             }
130         } 
131     }
